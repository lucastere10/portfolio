Here we’ll need to make use of some label encoding on categorical data

  #Label Encoding
  #drop unecessary columns
  drops = ['Row ID', 'Customer ID', 'Product ID','Country','Order Date','Ship Date','Discount','Profit']
  df_ml = df_ml.drop(drops, axis = 1)
  #look for categorical variables
  obj = (df_ml.dtypes == 'object')
  #Transform Categorical Data
  LE = LabelEncoder()
  obj_cols = list(obj[obj].index)
  for i in obj_cols:
     df_ml[i]=df_ml[[i]].apply(LE.fit_transform)
